selector.ebShow();

  • 특정 요소를 보여줄 때 사용하는 메소드
  • 셀렉터에 .show라는 클래스가 없어야만 실행 가능
  • 개요
    • 메소드 실행 시 셀렉터에 .show 클래스가 자동으로 붙음
    • 애니메이션 관련된 기본 css가 없으므로, 애니메이션이 필요할 경우 .show 클래스를 이용한 transition을 직접 입력해야 함

}

selector.ebPlayFaceid();
selector.ebResetFaceid();

  • face id 연동 화면을 재생
  • 기본 html/관련 css 템플릿 제공
  • ebPlayFaceid()
    • 커스텀하여 사용하고 싶을 경우 asset/img/faceid 폴더 안에 있는 faceid_img gif를 교체
    • ebPlayFaceid 함수 자체는 face id gif 재생에 관련된 함수기에 face id가 포함된 화면의 제어(ex. 보이기/닫기/dim...)는 별도로 작업해야 함(아래 예시 참고)
    ebPlayFaceid({onAfterPlay})
    • face id 플레이가 완료된 후 실행할 함수
  • ebResetFaceid()
    • 재생된 face id를 리셋시켜주는 함수

<!-- 버튼 -->
<button class="btn btn16" data-function="btn16">face id 실행</button>

<!-- face id 템플릿 -->
<div class="" data-type="faceid" data-name="페이스아이디">
    <span class="img_box"><img src="" alt="" /></span><!-- faceid gif -->
    <span class="txt_box"></span>
</div>



function btn16(){
    $('[data-name="페이스아이디"]').ebShow({
        onAfterShow: function() {
            $('[data-name=페이스아이디]').ebPlayFaceid({
                onAfterPlay: function() {
                    $('[data-name=페이스아이디]').ebHide({
                        onAfterHide: function() {
                            $('[data-name=페이스아이디]').ebResetFaceid();
                        }
                    })
                }
            })
        }
    })
}

selector.WSlot();

  • ios 스타일로 커스텀 된 select를 불러오는 함수. 기존에 있던 WSlot 플러그인을 제안 형태에 맞게 커스텀함.
  • 선택된 value가 들어가는 마크업만 있으면 셀렉트 리스트는 자동으로 append 됨.
  • 순서
    1. items로 셀렉트 리스트 생성
    2. WSlot('show'): 셀렉트 열기
    3. 셀렉트의 '확인'을 누르면 셀렉트가 닫히며 선택된 셀렉트로 value 변경
  • 셀렉터
    • data-type="select-ios"data-name="셀렉트 이름": value 영역
    • data-type="value": 선택된 셀렉트 옵션이 출력되는 곳. 기본 값 임력 필요
    • data-select-name: ios 스타일의 셀렉트 마크업 이름. data-name="셀렉트 이름"와 동일
    • .select-confirm: 리스트 속 확인 버튼. append로 생성한 마크업이기에 클래스 변경 불가
  • 옵션
    • WSlot('show')
      셀렉트 리스트 열기
    • WSlot('hide')
      셀렉트 리스트 닫기
    • WSlot('clear')
      data-value에 입력된 값을 초기값으로 되돌리기
    • WSlot('get')
      현재 선택된 항목의 index 가져오기
    • WSlot('getText')
      현재 선택된 항목의 텍스트 가져오기
    • WSlot('rollTo', index)
      적힌 순번으로 셀렉트 이동시킴
      주의사항: index 기입 시 첫번째 index는 -1, 두번째 부터 1,2,3.. 으로 인식함(0이 존재하지 않음)
  • 유의사항
    • ios 스타일이 입혀진 셀렉트 리스트를 변경하고 싶을 경우 css 직접 수정
    • data-type="select-ios" data-name="셀렉트 이름"은 value가 들어갈 영역이며 html 작성 필요,
      data-select-name="셀렉트 이름"은 자동으로 생성되는 마크업. 필요에 따라 셀렉터를 정확하게 지정할 것
    • .select-confirm을 누르고 나서의 함수를 지정할 경우 : 셀렉트 리스트는 마크업이 append되는 형태이므로 윈도우가 로드 된 후에 작성해야 함
    • 셀렉트 리스트가 열리는 순간, 다른 오브젝트의 클릭을 방지하기 위해 셀렉트 위에 투명한 레이어가 after로 덮어씌워져 있음
    • 기타 플러그인 참고사항 : http://danartinho.github.io/WSlot/

<!-- sample 1 -->
<div data-type="select-ios" data-name="셀렉트01">
    <div data-type="value">옵션</div>
</div>



.greenyellow { background-color: greenyellow; } 
    


select();
// 셀렉트 리스트 생성
function select(){
    $('[data-name="셀렉트01"]').WSlot({
        items:['select01','select02','select03','select04','select05'], // 셀렉트 리스트
        center: 4  // 처음 선택될 index
    });
}
// 셀렉트 열기
function btn20(){
    $('[data-name="셀렉트01"]').WSlot('show');
};
// 셀렉트 닫기
function btn21(){
    $('[data-name="셀렉트01"]').WSlot('hide');
};
// 셀렉트 2번째로 이동
function btn22(){
    $('[data-name="셀렉트01"]').WSlot('rollTo', 1);
}
// 셀렉트 1번째로 이동
function btn23(){
    $('[data-name="셀렉트01"]').WSlot('rollTo', -1);
}
// 셀렉트 초기화
function selectReset(){
    $('[data-select-name="셀렉트01"]').remove();
    $('[data-name="셀렉트01"]').WSlot({
        items:['select01','select02','select03','select04','select05'], // 셀렉트 리스트
        center: 4  // 처음 선택될 index
    });
    $('[data-name="셀렉트01"]').WSlot('clear');
};
// 확인 버튼 눌렀을 때
$(window).on('load', function(){
    $('[data-select-name="셀렉트01"] .select-confirm').on('click', function(){
        $('[data-name="셀렉트01"] [data-type="value"]').addClass('greenyellow');
    })
})
